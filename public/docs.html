const axios = require('axios');

class ZenithClient {
  constructor(apiKey) {
    this.apiKey = apiKey;
    this.baseURL = 'https://api.zenith.ai/v1';
  }

  async chat(message, model = 'claude-3.5') {
    try {
      const response = await axios.post(`${this.baseURL}/chat`, {
        message,
        model,
        stream: false
      }, {
        headers: {
          'X-API-Key': this.apiKey,
          'Content-Type': 'application/json'
        }
      });

      return response.data;
    } catch (error) {
      throw new Error(`Zenith API Error: ${error.response?.data?.message || error.message}`);
    }
  }

  async executeCommand(command, options = {}) {
    try {
      const response = await axios.post(`${this.baseURL}/terminal`, {
        command,
        ...options
      }, {
        headers: {
          'X-API-Key': this.apiKey,
          'Content-Type': 'application/json'
        }
      });

      return response.data;
    } catch (error) {
      throw new Error(`Terminal execution error: ${error.response?.data?.message || error.message}`);
    }
  }
}

// Uso
const zenith = new ZenithClient('your-api-key-here');

// Chat con IA
const chatResponse = await zenith.chat(
  'Create a REST API endpoint for user management',
  'gpt-4'
);
console.log(chatResponse.response);

// Ejecutar comando
const commandResult = await zenith.executeCommand(
  'Create a new React component with TypeScript'
);
console.log(commandResult.output);
